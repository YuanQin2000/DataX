444 ErrorCode CHttpRequest::HandleStatusCode(int statusCode, CHeaderField* pRespHeaderField)
445 {
446  if (statusCode < SC_MIN || statusCode > SC_MAX) {

448  return EC_PROTOCOL_ERROR;
449  }
450  m_StatusCode = statusCode;
451 
452  ErrorCode error = EC_NONE;
453  switch (statusCode) {
454  case SC100_CONTINUE:
455  case SC101_SWITCH_PROTOCOL:
456  ASSERT(false);
457  break;
458  case SC200_OK:
459  break;
460  case SC201_CREATED:
461  error = Handle201Created(pRespHeaderField);
462  break;
463  case SC202_ACCEPTED:
464  m_Client.OnHttpIndication(IND_HTTP_REQUEST_PROCESSING);
465  break;
466  case SC203_NON_AUTHORITATIVE:
467  // The information contained in the entity headers came not from the
468  // origin server but from a copy of the resource. This could happen
469  // if an intermediary had a copy of a resource but could not or did not
470  // validate the meta-information (headers) it sent about the resource.
471  //
472  // This response code is not required to be used; it is an option for
473  // applications that have a response that would be a 200 status if the
474  // entity headers had come from the origin server.
475  break;
476  case SC204_NO_CONTENT:
477  case SC205_RESET_CONTENT:
478  // Notify the status only, no content.
479  // 205 told the browser need to refresh the input form.
480  break;
481  case SC206_PARTIAL_CONTENT:
482  // For Range Request.
483  error = Handle206PartialContent(pRespHeaderField);
484  break;
485  case SC300_MULTIPLE_CHOICE:
486  // The choice list is in the payload.
487  break;
488  case SC301_MOVE_PERMANENTLY:
489  error = HandleRedirect(pRespHeaderField, true);
490  break;
491  case SC302_FOUND:
492  case SC307_TEMPORARY_REDIRECT:

495  case SC303_SEE_OTHER:
